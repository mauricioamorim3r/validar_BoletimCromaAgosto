### 3. `deploy-mcp.js` - Deploy Autom√°tico

#!/usr/bin/env node
/**
 * üöÄ DEPLOY AUTOM√ÅTICO VIA MCP
 * Deploy qualquer projeto para Render usando MCPs
 */
const fs = require('fs');
const { execSync } = require('child_process');

function loadProjectConfig() {
    if (!fs.existsSync('mcp-project-config.json')) {
        console.log('‚ùå Execute primeiro: node setup-mcp-project.cjs');
        process.exit(1);
    }
    return JSON.parse(fs.readFileSync('mcp-project-config.json', 'utf8'));
}

function detectDeployFiles() {
    const config = loadProjectConfig();
    const required = config.files || [];
    const missing = required.filter(file => !fs.existsSync(file));
    
    if (missing.length > 0) {
        console.log(`‚ö†Ô∏è  Arquivos ausentes: ${missing.join(', ')}`);
        return false;
    }
    return true;
}

function generateRenderConfig(config) {
    const renderYaml = `
services:
  - type: web
    name: ${path.basename(process.cwd())}
    runtime: ${config.runtime}
    buildCommand: ${config.buildCommand}
    startCommand: ${config.startCommand}
    envVars:
      - key: NODE_ENV
        value: production
      - key: PYTHON_VERSION
        value: "3.11.7"
`.trim();

    fs.writeFileSync('render.yaml', renderYaml);
    console.log('‚úÖ render.yaml criado');
}

function commitAndPush() {
    try {
        execSync('git add .');
        execSync('git commit -m "Deploy: Configura√ß√£o MCP autom√°tica"');
        execSync('git push');
        console.log('‚úÖ C√≥digo enviado para GitHub');
    } catch(e) {
        console.log('‚ö†Ô∏è  Erro no Git:', e.message);
    }
}

function main() {
    console.log('üöÄ INICIANDO DEPLOY AUTOM√ÅTICO...\n');
    
    const config = loadProjectConfig();
    console.log(`üìÇ Projeto: ${config.projectType.toUpperCase()}`);
    
    if (!detectDeployFiles()) {
        console.log('‚ùå Corrija os arquivos ausentes');
        return;
    }
    
    generateRenderConfig(config);
    commitAndPush();
    
    console.log('\nüéØ DEPLOY INICIADO!');
    console.log('üìä Monitore via Claude Desktop:');
    console.log('‚Ä¢ "Show deployment status"');
    console.log('‚Ä¢ "Check logs for this project"');
    console.log('‚Ä¢ "List my Render services"');
}

if (require.main === module) {
    main();
}
